\section{Resumen del avance}
\begin{frame}{Resumen del avance}
  \justifying
  \begin{itemize}
    \item Tema central: guia virtual inteligente para museos y espacios culturales.
    \item Problema abordado: mediacion estatica y baja personalizacion en visitas presenciales.
    \item Enfoque propuesto: aplicacion movil con BLE, TTS, STT e IA contextual.
  \end{itemize}
\end{frame}

\section{Problema y oportunidad}
\begin{frame}{Problema y oportunidad}
  \begin{block}{Problema}
    Los visitantes suelen depender de cartelas, recorridos no adaptativos o apoyo humano limitado.
  \end{block}
  \begin{block}{Oportunidad}
    Una guia inteligente puede activar contenido segun ubicacion y responder preguntas en tiempo real.
  \end{block}
\end{frame}

\section{Propuesta}
\begin{frame}{Propuesta MuseIQ}
  \begin{itemize}
    \item Localizacion indoor con BLE y beacons.
    \item Narracion automatica mediante TTS.
    \item Interaccion por voz con STT.
    \item Respuestas contextuales con IA y recuperacion de informacion.
  \end{itemize}
\end{frame}
